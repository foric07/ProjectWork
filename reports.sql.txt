-- Отчёт 1: Воронка продаж с детализацией по менеджерам
SELECT 
    e.full_name AS manager,
    ps.name AS stage,
    COUNT(d.id) AS deals_qty,
    SUM(d.amount) AS total_amount
FROM deals d
JOIN employees e ON d.responsible_employee_id = e.id
JOIN pipeline_stages ps ON d.stage_id = ps.id
WHERE d.status = 'open'
GROUP BY e.full_name, ps.name, ps.order_num
ORDER BY e.full_name, ps.order_num;

-- Отчёт 2: Эффективность сотрудников (расширенный, с KPI)
SELECT 
    e.full_name,
    COUNT(d.id) AS total_deals,
    COUNT(d.id) FILTER (WHERE d.status = 'won') AS won,
    ROUND(COUNT(d.id) FILTER (WHERE d.status = 'won') * 100.0 / NULLIF(COUNT(d.id), 0), 1) AS win_rate,
    COALESCE(SUM(d.amount) FILTER (WHERE d.status = 'won'), 0) AS revenue,
    COALESCE(AVG(EXTRACT(DAY FROM (d.actual_close_date - d.created_at))) FILTER (WHERE d.status = 'won'), 0) AS avg_days_to_close
FROM employees e
LEFT JOIN deals d ON d.responsible_employee_id = e.id
    AND d.created_at >= '2026-01-01'
GROUP BY e.id, e.full_name
ORDER BY revenue DESC;
